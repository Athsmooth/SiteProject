document.getElementById('launchBtn').onclick = async function() { // 1. Open the window const win = window.open("about:blank", "_blank"); if (!win) { alert("Popup blocked! Please allow popups."); return; } try { // 2. Fetch the library with cache-buster const response = await fetch('./index.html?v=' + Date.now()); if (!response.ok) throw new Error("Could not fetch index.html"); const html = await response.text(); // 3. Parse and Inject const parser = new DOMParser(); const doc = parser.parseFromString(html, 'text/html'); win.document.head.innerHTML = doc.head.innerHTML; win.document.body.innerHTML = doc.body.innerHTML; // 4. Re-run Scripts (Crucial for Cloudflare) const scripts = win.document.querySelectorAll("script"); scripts.forEach(oldScript => { const newScript = win.document.createElement("script"); // Copy attributes (src, type, etc.) Array.from(oldScript.attributes).forEach(attr => { newScript.setAttribute(attr.name, attr.value); }); // Copy internal code if it's not an external file if (oldScript.innerHTML) { newScript.appendChild(win.document.createTextNode(oldScript.innerHTML)); } oldScript.parentNode.replaceChild(newScript, oldScript); }); // 5. Close launcher after a short delay setTimeout(() => { window.close(); }, 500); } catch (err) { console.error("Launch Error:", err); if (win) { win.document.body.innerHTML = `

Error: ${err.message}

`; } } }; // End of onclick function
